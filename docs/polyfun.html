_⇒_ : ∀ {I : Set} → (I → Set) → (I → Set) → Set
X ⇒ Y = ∀ {i} → X i → Y i

data hvec : list Set → Set where
  [] : hvec []
  _∷_ : ∀ {T U} (x : T) (xs : hvec U) → hvec (T ∷ U)

collect : ∀ {ℓ} {A : Set ℓ} {n} → (fin n → A) → list A
collect {n = ze}   xs = []
collect {n = su n} xs = xs ze ∷ collect (xs ∘ su)

hvec′ : (n : ℕ) → (fin n → Set) → Set
hvec′ n T = hvec (collect T)

mk-hvec′ : ∀ {n} {T : fin n → Set} (xs : ∀ i → T i) → hvec′ n T
mk-hvec′ {n = ze}   xs = []
mk-hvec′ {n = su n} xs = xs ze ∷ mk-hvec′ (xs ∘ su)

hget′ : ∀ {n} {T : fin n → Set} (i : fin n) → hvec′ n T → T i
hget′ ze     (x ∷ _)  = x
hget′ (su i) (_ ∷ xs) = hget′ i xs

hfoldr′ : ∀ {n} {T : fin n → Set} (A : Set)
         → A → (∀ {i} → T i → A → A) → hvec′ n T → A
hfoldr′ {n = ze} A a f [] = a
hfoldr′ {n = su n} A a f (x ∷ xs) = f x (hfoldr′ A a f xs)

hmap′ : ∀ {n} {S T : fin n → Set} → (S ⇒ T) → hvec′ n S → hvec′ n T
hmap′ {n = ze} f [] = []
hmap′ {n = su n} f (x ∷ xs) = f x ∷ hmap′ f xs


record poly (I : Set) : Set₁ where
  field
    sh : I → Set
    po : ∀ {i} → sh i → ℕ
    nx : ∀ {i} (s : sh i) → fin (po s) → I
open poly public

record _⇒ₚ_ {I} (P Q : poly I) : Set where
  field
    sh⇒ : P .sh ⇒ Q .sh
    po⇐ : ∀ {i} (s : P .sh i) → fin (Q .po (sh⇒ s)) → fin (P .po s)
    coh : ∀ {i} {s : P .sh i} n → P .nx s (po⇐ s n) ≡ Q .nx _ n
open _⇒ₚ_ public


module _ {I : Set} where

  ⟦_⟧ₚ : poly I → (I → Set) → (I → Set)
  ⟦ P ⟧ₚ X i = Σ[ s ∈ P .sh i ] hvec′ (P .po s) (X ∘ P .nx s)

  ⟦_⟧⇒ₚ : ∀ {P Q} (m : P ⇒ₚ Q) {X} → ⟦ P ⟧ₚ X ⇒ ⟦ Q ⟧ₚ X
  ⟦ m ⟧⇒ₚ {X} (s , k) =
    m .sh⇒ s , mk-hvec′ (λ i → subst≡ X (m .coh i) (hget′ (m .po⇐ s i) k))

  fmapₚ : ∀ {P X Y} → X ⇒ Y → ⟦ P ⟧ₚ X ⇒ ⟦ P ⟧ₚ Y
  fmapₚ f (s , k) = s , hmap′ f k 


  _+ₚ_ : poly I → poly I → poly I
  (P +ₚ Q) .sh i = P .sh i ⊎ Q .sh i
  (P +ₚ Q) .po (inj₁ x) = P .po x
  (P +ₚ Q) .po (inj₂ y) = Q .po y
  (P +ₚ Q) .nx (inj₁ x) n = P .nx x n
  (P +ₚ Q) .nx (inj₂ y) n = Q .nx y n

  _×ₚ_ : poly I → poly I → poly I
  (P ×ₚ Q) .sh i = P .sh i × Q .sh i
  (P ×ₚ Q) .po (x , y) = P .po x + Q .po y
  (P ×ₚ Q) .nx (x , y) n with splitAt (P .po x) n
  ... | inj₁ n′ = P .nx x n′
  ... | inj₂ n′ = Q .nx y n′

  κ : (I → Set) → poly I
  κ X .sh = X
  κ X .po _ = 0
  κ X .nx s ()
